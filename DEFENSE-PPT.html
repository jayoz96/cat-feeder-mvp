// src/lib/distance.ts export function haversineDistance(a: Coordinates, b: Coordinates): number { const toRad = (deg: number) => (deg * Math.PI) / 180; const dLat = toRad(b.lat - a.lat); const dLng = toRad(b.lng - a.lng); const sinLat = Math.sin(dLat / 2); const sinLng = Math.sin(dLng / 2); const h = sinLat * sinLat + Math.cos(toRad(a.lat)) * Math.cos(toRad(b.lat)) * sinLng * sinLng; return 2 * EARTH_RADIUS_KM * Math.asin(Math.sqrt(h)); } export function formatDistance(km: number): string { return km < 1 ? `${Math.round(km*1000)}m` : `${km.toFixed(1)}km`; }

Code Review — 三层门禁机制

整个项目的 Code Review 由三层机制组成：

🔍 第一层：tsc --noEmit

每次改动后运行
捕获类型错误、变量遗漏

速度快，秒级反馈

🏗️ 第二层：npm run build

功能完成后运行
验证模块解析、Suspense 边界

比 tsc 更严格

👤 第三层：人工验收

UI/UX 体验审核
视觉层级、交互细节

主观体验反馈

形成闭环：编码 → tsc（类型安全）→ build（模块解析）→ 人工验收 → 反馈修复 → 编码

function autoCompleteExpired() { const now = Date.now(); for (const order of getStore()) { if (order.status === "pending_review" && order.completedAt && now - new Date(order.completedAt).getTime() >= AUTO_COMPLETE_MS) { order.status = "paid"; } } }

维度六

安全 — 五层防护体系

即使是 MVP 项目，安全边界从第一天就建立

🔑 1. API Key 防泄露

  • 高德 Key 通过 NEXT_PUBLIC_AMAP_KEY 环境变量注入
  • .env.local 被 .gitignore 排除
  • 密钥永远不会进入 Git 仓库

🛡️ 2. 优雅降级，不暴露错误

  • API 调用 try/catch 包裹
  • 失败时静默降级到本地地址库
  • 不向前端暴露错误堆栈或 API 细节

👤 3. 权限校验

  • deleteOrder 校验 userId 归属
  • 防止用户 A 删除用户 B 的订单
  • 后续接入真实认证无需重构

✅ 4. 输入校验 + 状态校验

  • Server Action 入口校验必填参数
  • 统一 ActionResult 返回结构
  • 每个状态变更检查当前状态合法性

安全 — 代码级防护

权限校验：防止越权删除

// src/services/order-service.ts:160-167 deleteOrder(orderId: string, userId: string): boolean { const store = getStore(); const order = store.find((o) => o.id === orderId); if (!order || order.userId !== userId) return false; // 权限校验 setStore(store.filter((o) => o.id !== orderId)); return true; }

Server Action 输入校验：统一错误返回

// src/app/(dashboard)/user/actions.ts export async function deleteOrder(orderId: string): Promise<ActionResult> { if (!orderId) return { success: false, message: "订单 ID 不能为空" }; try { const deleted = OrderService.deleteOrder(orderId, "user-1"); if (!deleted) return { success: false, message: "订单不存在或无法删除" }; return { success: true, message: "订单已删除" }; } catch { return { success: false, message: "删除失败，请稍后重试" }; // 不暴露内部错误 } }

安全 — Claude Code /security-review 命令

Claude Code 内置了 /security-review 命令，可对整个项目或指定文件进行安全审计：

> /security-review Claude Code 会自动扫描项目代码，检查以下安全维度： • 敏感信息泄露（API Key、密码硬编码） • 注入漏洞（XSS、SQL 注入、命令注入） • 不安全的依赖和配置 • 权限校验缺失 • 错误信息过度暴露

在本项目中的应用

🔍 审查发现

  • 确认 .env.local 已被 .gitignore 排除
  • 确认 Server Action 的 catch 不暴露堆栈
  • 确认状态流转有前置状态校验
  • 确认 deleteOrder 有 userId 归属检查

💡 审查建议

  • 建议对用户输入做进一步 sanitize
  • 建议添加 rate limiting 防刷
  • 建议 NEXT_PUBLIC_ 前缀 Key 注意前端可见性
  • MVP 阶段已有完善安全基线

/security-review 相当于一次自动化的安全 Code Review，是 tsc + build 之外的第三道安全门禁
